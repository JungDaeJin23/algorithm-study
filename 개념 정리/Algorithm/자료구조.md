# 자료구조
## Stack
```python
class Stack:
    def __init__(self, size):
        self.__size = size
        self.__stack = [None] * size
        self.__top = -1
        
    def top(self):
        if self.empty():
            raise Exception("Stack is empty")
        return self.__stack[self.__top]
    
    def empty(self):
        if self.__top == -1:
            return True
        return False
    
    def pop(self):
        if self.empty():
            raise Exception("Stack is empty")
        top_value = self.__stack[self.__top]
        self.__stack[self.__top] = None
        self.__top -= 1
        return top_value
    
    def push(self, el):
        if self.full():
            raise Exception("Stack is full")
        self.__top += 1
        self.__stack[self.__top] = el
        return
    
    def full(self):
        if self.__top + 1 == self.__size:
            return True
        return False
        
    def __repr__(self):
        return f"{self.__stack}"
```
